					</list>
				</property>
			-->	
			</bean>
		</mvc:interceptor>

			<bean class="com.ccms.interceptor.Log4LoginInterceptor">
				<property name="accessLogService" ref="accessLogService"></property>
			</bean>
		</mvc:interceptor>
